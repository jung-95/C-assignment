//배열의 특정 구간에서 최솟값 찾아 옮기는 함수

#include <stdio.h>

//배열 데이터, 변수 선언
int A[10] = { 5,8,2,6,1,3,4,10,9,7 };
int min, temp;

//최솟값을 찾고, A[b]와 자리를 바꾸는 함수
void findMin(int b, int e)
{
	int i;
	min = b;
	for (i = b; i <= e; i++)
	{
		if (A[min] > A[i])
		{
			min = i;
		}
	}

	temp = A[b];
	A[b] = A[min];
	A[min] = temp;

}

//배열을 나열해 표시하는 함수
void printArray(int n)
{
	int i;
	for (i = 0; i <= n - 1; i++)
	{
		printf("%d ", A[i]);
	}

	printf("\n");
}

int main(void)
{
	int n = 10;
	printf("[0] : ");
	printArray(n);
	findMin(0, n - 1);

	printf("[1] : ");
	printArray(n);
	findMin(1, n - 1);

	printf("[2] : ");
	printArray(n);
	findMin(2, n - 1);

	printf("[3] : ");
	printArray(n);
	findMin(3, n - 1);

	return 0;
}
