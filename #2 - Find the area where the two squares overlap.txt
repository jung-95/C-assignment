//(문제2) 두 사각형이 겹치는 부분의 면적 계산하기

#include <stdio.h>
int main(void)
{
	//사각형 R1, R2 좌표 정하기
	int x1, x2, x3, x4, x5, x6, y1, y2, y3, y4, y5, y6;
	printf("사각형1의 좌표 입력 : ");
	scanf_s("%d %d %d %d", &x1, &y1, &x2, &y2);
	printf("사각형2의 좌표 입력 : ");
	scanf_s("%d %d %d %d", &x3, &y3, &x4, &y4);

	//겹치지 않는 경우
	if (x3 >= x2 || x4 <= x1)
	{
		printf("겹치지 않는다.");
	}

	else if (y4 <= y1 || y3 >= y2)
	{
		printf("겹치지 않는다.\n\n");
	}

	//겹치는 경우
	else
	{
		//겹치는 부분 사각형 R3 좌표 정하기(x5, y5),(x6,y6)
		//x5 정하기
		if (x1 <= x3)
		{
			x5 = x3;
		}

		else
		{
			x5 = x1;
		}

		//x6 정하기
		if (x2 <= x4)
		{
			x6 = x2;
		}

		else
		{
			x6 = x4;
		}

		//y6 정하기
		if (y2 <= y4)
		{
			y6 = y2;
		}

		else
		{
			y6 = y4;
		}

		//y5 정하기
		if (y1 <= y3)
		{
			y5 = y3;
		}

		else
		{
			y5 = y1;
		}

		//겹치는 부분의 좌표와 면적 나타내기
		printf("겹치는 부분의 좌표 = (%d, %d), (%d, %d)\n", x5, y5, x6, y6);
		printf("겹치는 부분의 면적 = %d", (x6 - x5) * (y6 - y5));
	}

	return 0;
}
