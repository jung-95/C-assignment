#include <stdio.h>
int main(void)
{
	int i, j, n;
	printf("n = ");
	scanf_s("%d", &n);

	for (i = 0; i < n * n; i++)
	{ 
		if ((i / n) % 2 == 0)
		{
			for (j = 0; j < n * n; j++)
			{
				if ((j / n) % 2 == 0)
				{
					printf("0");
				}

				else
				{
					printf("X");
				}
			}
		
		}

		else
		{
			for (j = 0; j < n * n; j++)
			{
				if ((j / n) % 2 != 0)
				{
					printf("0");
				}

				else
				{
					printf("X");
				}
			}
		}
		printf("\n");
	}
	return 0;
}
